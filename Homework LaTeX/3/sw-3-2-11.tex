\gdef\problemNumber{SW 3.2.11}
\def\problemText{
How many binary tree shapes of $N$ nodes are there with height $N$? How many
different ways are there to insert $N$ distinct keys into an initially empty BST
that result in a tree of height $N$?\\[12pt]
}
\def\problemSolution{
\textcolor{blue}{
\textbf{Answer:}\\[6pt]
$2^{N - 1}$ ways. \\[6pt]
If $N > 1$, we have two choices at each node: either to insert on the left or right. \\[6pt]
}\\[6pt]
}
