\gdef\problemNumber{GT R-1.7}
\def\problemText{
\setcounter{myenumi}{\value{enumi}}
Order the following list of functions by the $O(\,)$ notation, from smallest to
largest. Group together those functions that are in $\Theta(\,)$ of one another.
\\[12pt]
\begin{tabbing}
    $6n\lg n$~~~~~~~~~~ \= $2^{100}$~~~~~~~~~~ \= $\log\log n$~~~~~~~~~~ \= $\log^2n$~~~~~~~~~~ \= $2^{\log n}$\\[6pt]
    $2^{2^n}$ \> $\lceil\sqrt n\rceil$ \> $n^{0.01}$ \> $1/n$ \> $4n^{3/2}$\\[6pt]
    $3n^{0.5}$ \> $5n$ \> $\lfloor2n\log^2n\rfloor$ \> $2^n$ \> $n\log_4n$\\[6pt]
    $4^n$ \> $n^3$ \> $n^2\log n$ \> $4^{\log n}$ \> $\sqrt{\log n}$\\[6pt]
\end{tabbing}
\vskip12pt
\emph{Note}\/: If no base is given, assume that $\log n$ is $\log_2 n$.\\[12pt]
\emph{Hint}\/: When in doubt about two functions $f(n)$ and $g(n)$, consider
$\log f(n)$ and $\log g(n)$ or $2^{f(n)}$ and $2^{g(n)}$.\\[12pt]
\setcounter{enumi}{\value{myenumi}}
}
\def\problemSolution{
\textcolor{blue}{
\textbf{Answer:}\\[6pt]
\begin{enumerate} 
    \item $1/n$ 
    \item $2^{100}$ : $\Theta(1)$ 
    \item $\log\log n$ : $\Theta(\log\log n)$
    \item $\sqrt{\log n}$ : $\Theta(\log n)$
    \item $\log^2n$ : $\Theta(\log n)$
    \item $n^{0.01}$ : $\Theta(n^{0.01})$ 
    \item $\lceil\sqrt n\rceil$, $3n^{0.5}$ : $\Theta(n^{1/2})$
    \item $5n$ : $\Theta(n)$
    \item $2^{\log n}$ : $\Theta(n)$ 
    \item $6n\lg n$, $n\log_4n$ : $\Theta(n\log n)$ 
    \item $\lfloor2n\log^2n\rfloor$ : $\Theta(n\log^2n)$
    \item $4n^{3/2}$ : $\Theta(n^{3/2})$ 
    \item $4^{\log n}$ : $\Theta(n^2)$ 
    \item $n^2\log n$ : $\Theta(n^2\log n)$ 
    \item $n^3$ : $\Theta(n^3)$ 
    \item $2^n$ : $\Theta(2^n)$ 
    \item $2^{2^n}$, $4^n$ : $\Theta(4^n)$ 
\end{enumerate}
}

}
