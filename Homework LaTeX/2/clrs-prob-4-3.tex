\gdef\problemNumber{CLRS Problem 4-3 (modified)}
\def\problemText{
Give an asymptotically tight bound for $T(n)$ in each of the following
recurrences.
\begin{list}{\textbf{\alph{enumii}}}{\usecounter{enumii}}
  \item$T(n)=4T\left(n/3\right)+n\lg n$
  \item$T(n)=4T\left(n/2\right)+n^2\sqrt{n}$
  \item$T(n)=T\left(n/2\right)+T\left(n/4\right)+T\left(n/8\right)+n$
  \item$T(n)=T\left(n-1\right)+1/n$
  \item$T(n)=T\left(n-1\right)+\lg n$
\end{list}
}
\def\problemSolution{
\textcolor{blue}{
\textbf{Answer:}\\
\\[6pt]a. $T(n)=4T\left(n/3\right)+n\lg n$ \\[6pt]
Here: a = 4, b = 3, f(n) = n lg n \\[6pt]
So $lg_ba = lg_34 \approx 1.26$ \\[6pt]
Therefore, Case 1 says: $ f(n) = n lg n $ is smaller than $n^{1.26}$ \\[6pt]
$T(n) = \Theta(n^{1.26})$\\[12pt]
\\[6pt]b. $T(n)=4T\left(n/2\right)+n^2\sqrt{n}$\\[6pt]
Here: a = 4, b = 2, $f(n) = n^2 \sqrt{n}$ \\[6pt]
So $lg_ba = lg_24 = 2$ \\[6pt]
Therefore, Case 3 says: $ f(n) =  n^2 \sqrt{n} $ is greater than $n^2$ \\[6pt]
$T(n) = \Theta(n^2 \sqrt{n})$\\[12pt]
\\[6pt]c. $T(n)=T\left(n/2\right)+T\left(n/4\right)+T\left(n/8\right)+n$\\[6pt]
Using recurrence tree:\\[6pt]
$T(n)=n + T\left(n/2\right)+T\left(n/4\right)+T\left(n/8\right)$\\[6pt]
\begin{align*}
T(n) &= n + T\left(\frac{n}{2}\right) + T\left(\frac{n}{4}\right) + T\left(\frac{n}{8}\right) \\
     &= n + \left[ \frac{n}{2} + T\left(\frac{n}{4}\right) + T\left(\frac{n}{8}\right) + T\left(\frac{n}{16}\right) \right] \\
     &\quad + \left[ \frac{n}{4} + T\left(\frac{n}{8}\right) + T\left(\frac{n}{16}\right) + T\left(\frac{n}{32}\right) \right] \\
     &\quad + \left[ \frac{n}{8} + T\left(\frac{n}{16}\right) + T\left(\frac{n}{32}\right) + T\left(\frac{n}{64}\right) \right] \\
     &\quad\vdots \\
\end{align*}
At each level, the total work is a decreasing geometric sum:
\[
\text{Level 0: } n,\quad
\text{Level 1: } \frac{n}{2} + \frac{n}{4} + \frac{n}{8} = \frac{7n}{8},\quad
\text{Level 2: } \left( \frac{7}{8} \right)^2 n,\quad \ldots
\]
Total work across all levels:
\[
T(n) = n \left(1 + \frac{7}{8} + \left(\frac{7}{8}\right)^2 + \cdots \right)
= n \cdot \sum_{i=0}^{\infty} \left( \frac{7}{8} \right)^i
= n \cdot \frac{1}{1 - \frac{7}{8}} = 8n
\]
$T(n) = \Theta(n)$\\[6pt]
\\[6pt]d. $T(n)=T\left(n-1\right)+1/n$\\[6pt]
Using recurrence tree: \\[6pt]
$T(n)=1/n + T\left(n-1\right)$\\[6pt]
$T(n)= \frac{1}{n-1} + T\left(n-2\right)$\\[6pt]
$T(n)= \frac{1}{n-2} + T\left(n-3\right)$\\[6pt]
$T(n)= \frac{1}{n-3} + T\left(n-4\right)$\\[6pt]
Continue....\\[6pt]
$T(2)= \frac{1}{2} + T\left(1\right)$\\[6pt]
$T(1)= \frac{1}{1} + T\left(0\right)$\\[6pt]
We can see a pattern here: $T(n = )\frac{1}{1} + \frac{1}{2} + \frac{1}{3} \dots + \frac{1}{n}$\\[6pt]
This is harmonic series. We know that Harmonic Series is : $O(\lg n)$\\[6pt]
Therefore, $T(n) = \Theta(\lg n)$\\[6pt]
\\[6pt]e. $T(n)=T\left(n-1\right)+\lg n$\\[6pt]
Using recurrence tree: \\[6pt]
$T(n) = \lg n + T(n-1)$ \\[6pt]
$T(n-1) = \lg n - 1 + T(n-2)$ \\[6pt]
$T(n-2) = \lg n - 2 + T(n-3)$ \\[6pt]
$T(n-3) = \lg n - 3 + T(n-4)$ \\[6pt]
Continue ..... \\[6pt]
$T(2) = \lg 2 + T(1)$ \\[6pt]
$T(1) = \lg 1 + T(0)$ \\[6pt]
We can see a pattern here: $T(n) = \lg n + \lg n-1 + \lg n-2 + \dots + \lg 2 + \lg 1$. \\[6pt] 
$T(n) = \lg [n * (n-1) * (n-2) * \dots * 2 * 1]$. \\[6pt] 
Which is $\lg n!.$ We know that is: $\Theta(n \lg n)$\\[12pt] 
}
}
